def encode(text = input("Enter a text below please: ")):
    tx = ""
    text = text.lower()
    text = "".join(text.split())
    for x in range(0,len(text)):
        conv_char = ord(text[x]) - 96
        if conv_char > 0 and conv_char <= 26:
            tx += str(conv_char) + "."
        elif text[x] in [chr(c) for c in range(33,48)]:
            tx += '999.'
    return(tx)
